45 void GRPLHCIFfileProcessor::run(o2::framework::ProcessingContext& pc)
46 {
47  auto configBuff = pc.inputs().get<gsl::span<char>>("inputConfig");
48  auto configFileName = pc.inputs().get<std::string>("inputConfigFileName");
49  auto timer = std::chrono::duration_cast<std::chrono::milliseconds>(HighResClock::now().time_since_epoch()).count();
50  LOG(INFO) << "got input file " << configFileName << " of size " << configBuff.size();
51  mReader.loadLHCIFfile(configBuff);
52  std::vector<std::pair<long, std::vector<int32_t>>> beamEnergy;
53  std::vector<std::pair<long, std::vector<std::string>>> injScheme;
54  std::vector<std::pair<long, std::vector<std::string>>> fillNb;
55  std::vector<std::pair<long, std::vector<int32_t>>> atomicNbB1;
56  std::vector<std::pair<long, std::vector<int32_t>>> atomicNbB2;
57 
58  int nEleBeamEn, nEleInjSch, nEleFillNb, nEleAtNbB1, nEleAtNbB2 = 0;
59  int nMeasBeamEn, nMeasInjSch, nMeasFillNb, nMeasAtNbB1, nMeasAtNbB2 = 0;
60  std::string type{""};
61 
62  LHCIFData lhcifdata;
63 
64  mReader.readValue<int32_t>("BEAM_ENERGY", type, nEleBeamEn, nMeasBeamEn, beamEnergy);
65  if (nMeasBeamEn == 0) {
66  LOG(FATAL) << "Beam energy not present";
67  }
68  if (nEleBeamEn != 1 || nMeasBeamEn != 1) {
69  LOG(ERROR) << "More than one value/measurement found for Beam Energy, keeping the last one";
70  }
71  LOG(INFO) << "beam energy size = " << beamEnergy.size();
72  lhcifdata.setBeamEnergy(beamEnergy.back().first, beamEnergy.back().second.back());
73 
74  mReader.readValue<std::string>("INJECTION_SCHEME", type, nEleInjSch, nMeasInjSch, injScheme);
75  if (nMeasInjSch == 0) {
76  LOG(FATAL) << "Injection scheme not present";
77  }
78  if (nEleInjSch != 1 || nMeasInjSch != 1) {
79  LOG(ERROR) << "More than one value/measurement found for Injection Scheme, keeping the last one";
80  }
81  lhcifdata.setInjectionScheme(injScheme.back().first, injScheme.back().second.back());
82 
83  mReader.readValue<std::string>("FILL_NUMBER", type, nEleFillNb, nMeasFillNb, fillNb);
84  if (nMeasFillNb == 0) {
85  LOG(FATAL) << "Fill Number not present";
86  }
87  if (nEleFillNb != 1 || nMeasFillNb != 1) {
88  LOG(ERROR) << "More than one value/measurement found for Fill Number, keeping the last one";
89  }
90  lhcifdata.setFillNumber(fillNb.back().first, atoi(fillNb.back().second.back().c_str()));
91 
92  mReader.readValue<int32_t>("ATOMIC_NUMBER_B1", type, nEleAtNbB1, nMeasAtNbB1, atomicNbB1);
93  if (nMeasAtNbB1 == 0) {
94  LOG(FATAL) << "Atomic Number Beam 1 not present";
95  }
96  if (nEleAtNbB1 != 1 || nMeasAtNbB1 != 1) {
97  LOG(ERROR) << "More than one value/measurement found for Atomic Number Beam 1, keeping the last one";
98  }
99  lhcifdata.setAtomicNumberB1(atomicNbB1.back().first, atomicNbB1.back().second.back());
100 
101  mReader.readValue<int32_t>("ATOMIC_NUMBER_B2", type, nEleAtNbB2, nMeasAtNbB2, atomicNbB2);
102  if (nMeasAtNbB2 == 0) {
103  LOG(FATAL) << "Atomic Number Beam 2 not present";
104  }
105  if (nEleAtNbB2 != 1 || nMeasAtNbB2 != 1) {
106  LOG(ERROR) << "More than one value/measurement found for Atomic Number Beam 2, keeping the last one";
107  }
108  lhcifdata.setAtomicNumberB2(atomicNbB2.back().first, atomicNbB2.back().second.back());
109 
110  if (mVerbose) {
111  LOG(INFO) << " **** Beam Energy ****";
112  for (auto& el : beamEnergy) {
113  for (auto elVect : el.second) {
114  std::cout << el.first << " --> " << elVect << std::endl;
115  }
116  }
117  LOG(INFO) << " **** Injection Scheme ****";
118  for (auto& el : injScheme) {
119  for (auto elVect : el.second) {
120  std::cout << el.first << " --> " << elVect << std::endl;
121  }
122  }
123  LOG(INFO) << " **** Fill Number ****";
124  for (auto& el : fillNb) {
125  for (auto elVect : el.second) {
126  std::cout << el.first << " --> " << elVect << std::endl;
127  }
128  }
129  LOG(INFO) << " **** Atomic Number Beam 1 ****";
130  for (auto& el : atomicNbB1) {
131  for (auto elVect : el.second) {
132  std::cout << el.first << " --> " << elVect << std::endl;
133  }
134  }
135  LOG(INFO) << " **** Atomic Number B2 ****";
136  for (auto& el : atomicNbB2) {
137  for (auto elVect : el.second) {
138  std::cout << el.first << " --> " << elVect << std::endl;
139  }
140  }
141  }
142 
143  sendOutput(pc.outputs(), timer, lhcifdata);
144 }

154 void GRPLHCIFfileProcessor::sendOutput(DataAllocator& output, long tf, const LHCIFData& lhcifdata)
155 {
156  // sending output to CCDB
157 
158  constexpr uint64_t INFINITE_TF = 0xffffffffffffffff;
159 
160  using clbUtils = o2::calibration::Utils;
161  auto clName = o2::utils::MemFileHelper::getClassName(lhcifdata);
162  auto flName = o2::ccdb::CcdbApi::generateFileName(clName);
163  std::map<std::string, std::string> md;
164  md.emplace("created by", "dpl");
165  o2::ccdb::CcdbObjectInfo info("GRP/Data/LHCIFData", clName, flName, md, tf, INFINITE_TF);
166  auto image = o2::ccdb::CcdbApi::createObjectImage(&lhcifdata, &info);
167  LOG(INFO) << "Sending object " << info.getPath() << "/" << info.getFileName() << " of size " << image->size()
168  << " bytes, valid for " << info.getStartValidityTimestamp() << " : " << info.getEndValidityTimestamp();
169  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "GRP_LHCIFData", 0}, *image.get()); // vector<char>
170  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, "GRP_LHCIFData", 0}, info); // root-serialized
171 }
