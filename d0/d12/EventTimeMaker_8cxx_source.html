28 void computeEvTime(const std::vector<eventTimeTrack>& tracks, const std::vector<int>& trkIndex, eventTimeContainer& evtime)
29 {
30  const int maxNumberOfSets = 200;
31 
32  int ntracks = tracks.size();
33 
34  if (ntracks < 2) { // at least 2 tracks required
35  return;
36  }
37 
38  int hypo[MAXNTRACKINSET];
39 
40  int nmaxtracksinset = ntracks > 22 ? 6 : MAXNTRACKINSET; // max number of tracks in a set for event time computation
41  int ntracksinset = std::min(ntracks, nmaxtracksinset);
42 
43  Int_t nset = ((ntracks - 1) / ntracksinset) + 1;
44  int ntrackUsed = ntracks;
45 
46  if (nset > maxNumberOfSets) {
47  nset = maxNumberOfSets;
48  ntrackUsed = nmaxtracksinset * nset;
49  }
50 
51  // list of tracks in set
52  std::vector<int> trackInSet[maxNumberOfSets];
53 
54  for (int i = 0; i < ntrackUsed; i++) {
55  int iset = i % nset;
56 
57  trackInSet[iset].push_back(i);
58  }
59 
60  int status;
61  // compute event time for each set
62  for (int iset = 0; iset < nset; iset++) {
63  unsigned long bestComb = 0;
64  while (!(status = getStartTimeInSet(tracks, trackInSet[iset], bestComb))) {
65  ;
66  }
67  if (status == 1) {
68  int ntracks = trackInSet[iset].size();
69  // set the best in set
70  for (int itrk = 0; itrk < ntracks; itrk++) {
71  hypo[itrk] = bestComb % 3;
72  bestComb /= 3;
73 
74  int index = trkIndex[trackInSet[iset][itrk]];
75  const eventTimeTrack& ctrack = tracks[trackInSet[iset][itrk]];
76  evtime.weights[index] = 1. / (ctrack.expSigma[hypo[itrk]] * ctrack.expSigma[hypo[itrk]]);
77  evtime.tracktime[index] = ctrack.mSignal - ctrack.expTimes[hypo[itrk]];
78  }
79  }
80  } // end loop in set
81 
82  // do average among all tracks
83  float finalTime = 0, allweights = 0;
84  for (int i = 0; i < evtime.weights.size(); i++) {
85  if (evtime.weights[i] < 1E-6) {
86  continue;
87  }
88  allweights += evtime.weights[i];
89  finalTime += evtime.tracktime[i] * evtime.weights[i];
90  }
91 
92  if (allweights < 1E-6) {
93  return;
94  }
95 
96  evtime.eventTime = finalTime / allweights;
97  evtime.eventTimeError = sqrt(1. / allweights);
98 }

100 int getStartTimeInSet(const std::vector<eventTimeTrack>& tracks, std::vector<int>& trackInSet, unsigned long& bestComb)
101 {
102  float chi2, chi2best, averageBest = 0;
103  int hypo[MAXNTRACKINSET];
104  float starttime[MAXNTRACKINSET], weighttime[MAXNTRACKINSET];
105 
106  chi2best = 10000;
107  int ntracks = trackInSet.size();
108 
109  if (ntracks < 3) {
110  return 2; // no event time in the set
111  }
112 
113  unsigned long ncomb = combinatorial[ntracks];
114  for (unsigned long comb = 0; comb < ncomb; comb++) {
115  unsigned long curr = comb;
116 
117  int ngood = 0;
118  float average = 0;
119  float sumweights = 0;
120  // get track info in the set for current combination
121  for (int itrk = 0; itrk < ntracks; itrk++) {
122  hypo[itrk] = curr % 3;
123  curr /= 3;
124  const eventTimeTrack& ctrack = tracks[trackInSet[itrk]];
125  starttime[itrk] = ctrack.mSignal - ctrack.expTimes[hypo[itrk]];
126  weighttime[itrk] = 1. / (ctrack.expSigma[hypo[itrk]] * ctrack.expSigma[hypo[itrk]]);
127 
128  average += starttime[itrk] * weighttime[itrk];
129  sumweights += weighttime[itrk];
130  ngood++;
131  }
132 
133  average /= sumweights;
134 
135  // compute chi2
136  chi2 = 0;
137  float deltat;
138  for (int itrk = 0; itrk < ntracks; itrk++) {
139  deltat = starttime[itrk] - average;
140  chi2 += deltat * deltat * weighttime[itrk];
141  }
142  chi2 /= (ngood - 1);
143 
144  if (chi2 < chi2best) {
145  bestComb = comb;
146  chi2best = chi2;
147  averageBest = average;
148  }
149  } // end loop in combinations
150 
151  int worse = -1;
152  float errworse = 4;
153  // check the best combination
154  unsigned long curr = bestComb;
155  for (int itrk = 0; itrk < ntracks; itrk++) {
156  hypo[itrk] = curr % 3;
157  curr /= 3;
158 
159  const eventTimeTrack& ctrack = tracks[trackInSet[itrk]];
160  float err = ctrack.mSignal - ctrack.expTimes[hypo[itrk]] - averageBest;
161  err /= ctrack.expSigma[hypo[itrk]];
162  err = fabs(err);
163  if (err > errworse) {
164  errworse = err;
165  worse = itrk;
166  }
167  }
168 
169  if (worse > -1) {
170  const eventTimeTrack& ctrack = tracks[trackInSet[worse]];
171  // remove the track and try again
172  trackInSet.erase(trackInSet.begin() + worse);
173  return 0;
174  }
175 
176  return 1; // good event time in the set
177 }

179 void generateEvTimeTracks(std::vector<eventTimeTrackTest>& tracks, int ntracks, float evTime)
180 {
181  eventTimeTrackTest track;
182  constexpr float masses[3] = {0.13957000, 0.49367700, 0.93827200};
183  constexpr float kCSPEED = TMath::C() * 1.0e2f * 1.0e-12f;
184  float energy = 0.f;
185  float betas[3] = {0.f};
186 
187  float pMismatch = ntracks * 0.00005;
188 
189  for (int i = 0; i < ntracks; i++) {
190  track.mTOFChi2 = 1.f;
191  track.mP = gRandom->Exp(1);
192  track.mPt = track.mP;
193  track.mLength = 400.;
194  track.mHypo = gRandom->Exp(1);
195  if (track.mHypo > 2) {
196  track.mHypo = 2;
197  }
198  for (int j = 0; j < 3; j++) {
199  energy = sqrt(masses[j] * masses[j] + track.mP * track.mP);
200  betas[j] = track.mP / energy;
201  track.expTimes[j] = track.mLength / (betas[j] * kCSPEED);
202  track.expSigma[j] = 100.f;
203  if (j == track.mHypo) {
204  track.mSignal = track.expTimes[j] + gRandom->Gaus(0.f, track.expSigma[j]);
205 
206  if (gRandom->Rndm() < pMismatch) { // assign time from a different particle
207  float p = gRandom->Exp(1);
208  float l = 400;
209  int hypo = gRandom->Exp(1);
210  if (hypo > 2) {
211  hypo = 2;
212  }
213  energy = sqrt(masses[hypo] * masses[hypo] + p * p);
214  float beta = p / energy;
215  track.mSignal = l / (beta * kCSPEED) + gRandom->Gaus(0.f, track.expSigma[j]);
216  }
217  }
218  }
219  tracks.push_back(track);
220  }
221 }
