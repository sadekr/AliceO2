123 bool MatchITSTPCQC::selectTrack(o2::tpc::TrackTPC const& track)
124 {
125 
126  if (track.getPt() < mPtCut) {
127  return false;
128  }
129  if (std::abs(track.getEta()) > mEtaCut) {
130  return false;
131  }
132  if (track.getNClusters() < mNTPCClustersCut) {
133  return false;
134  }
135 
136  math_utils::Point3D<float> v{};
137  std::array<float, 2> dca;
138  if (!(const_cast<o2::tpc::TrackTPC&>(track).propagateParamToDCA(v, mBz, &dca, mDCACut)) || std::abs(dca[0]) > mDCACutY) {
139  return false;
140  }
141 
142  return true;
143 }

147 void MatchITSTPCQC::finalize()
148 {
149 
150  // first we use mPt and mPtTPC to set the TEfficiency; later they are scaled
151  if (!mFractionITSTPCmatch->SetTotalHistogram(*mPtTPC, "") ||
152  !mFractionITSTPCmatch->SetPassedHistogram(*mPt, "")) {
153  LOG(FATAL) << "Something wrong when defining the efficiency histograms!";
154  }
155 
156  float scaleFactTPC = 1. / mNTPCSelectedTracks;
157  float scaleFactITSTPC = 1. / mNITSTPCSelectedTracks;
158  mPtTPC->Scale(scaleFactTPC);
159  mPt->Scale(scaleFactITSTPC);
160  mEta->Scale(scaleFactITSTPC);
161  mChi2Matching->Scale(scaleFactITSTPC);
162  mChi2Refit->Scale(scaleFactITSTPC);
163  //mTimeResVsPt->Scale(scaleFactITSTPC); // if to few entries, one sees nothing after normalization --> let's not normalize
164 }
