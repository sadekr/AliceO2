113 template <class T>
114 inline std::string getType()
115 {
116  if (typeid(T).name() == typeid(char).name()) {
117  return std::string{"char"};
118  }
119  if (typeid(T).name() == typeid(size_t).name()) {
120  return std::string{"unsigned_long"};
121  }
122  if (typeid(T).name() == typeid(int).name()) {
123  return std::string{"int"};
124  }
125  if (typeid(T).name() == typeid(int4).name()) {
126  return std::string{"int4"};
127  }
128  return std::string{"unknown"};
129 }

154 inline float computeThroughput(Test test, float result, float chunkSizeGB, int ntests)
155 {
156  // https://docs.nvidia.com/cuda/cuda-c-best-practices-guide/index.html
157  // Eff_bandwidth (GB/s) = (B_r + B_w) / (~1e9 * Time (s))
158 
159  return 1e3 * chunkSizeGB * ntests / result;
160 }

273 inline void ResultWriter::addBenchmarkEntry(const std::string bName, const std::string type, const int nChunks)
274 {
275  mTimeTrees.emplace_back(new TTree((bName + "_" + type).data(), (bName + "_" + type).data()));
276  mTimeResults.clear();
277  mTimeResults.resize(nChunks);
278  mTimeTrees.back()->Branch("elapsed", &mTimeResults);
279 
280  mThroughputTrees.emplace_back(new TTree((bName + "_" + type + "_TP").data(), (bName + "_" + type + "_TP").data()));
281  mThroughputResults.clear();
282  mThroughputResults.resize(nChunks);
283  mThroughputTrees.back()->Branch("throughput", &mThroughputResults);
284 }
